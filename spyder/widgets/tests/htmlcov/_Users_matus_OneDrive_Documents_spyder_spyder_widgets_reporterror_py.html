264 def _submit_to_github(self): 

265 """Action to take when pressing the submit button.""" 

266 # Get reference to the main window 

267 if self.parent() is not None: 

268 if getattr(self.parent(), 'main', False): 

269 # This covers the case when the dialog is attached 

270 # to the internal console 

271 main = self.parent().main 

272 else: 

273 # Else the dialog is attached to the main window 

274 # directly 

275 main = self.parent() 

276 else: 

277 main = None 

278 

279 # Getting description and traceback 

280 title = self.title.text() 

281 description = self.input_description.toPlainText() 

282 traceback = self.error_traceback[:-1] # Remove last EOL 

283 

284 # Render issue 

285 if main is not None: 

286 issue_text = main.render_issue(description=description, 

287 traceback=traceback) 

288 else: 

289 issue_text = description 

290 

291 try: 

292 if main is None: 

293 org = 'ccordoba12' 

294 else: 

295 org = 'spyder-ide' 

296 github_backend = GithubBackend(org, 'spyder', parent_widget=main) 

297 github_report = github_backend.send_report(title, issue_text) 

298 if github_report: 

299 self.close() 

300 except Exception: 

301 ret = QMessageBox.question( 

302 self, _('Error'), 

303 _("An error occurred while trying to send the issue to " 

304 "Github automatically. Would you like to open it " 

305 "manually?<br><br>" 

306 "If so, please make sure to paste your clipboard " 

307 "into the issue report box that will appear in a new " 

308 "browser tab before clicking <i>Submit</i> on that " 

309 "page.")) 

310 if ret in [QMessageBox.Yes, QMessageBox.Ok]: 

311 QApplication.clipboard().setText(issue_text) 

312 issue_body = ( 

313 " \n<!--- *** BEFORE SUBMITTING: PASTE CLIPBOARD HERE " 

314 "TO COMPLETE YOUR REPORT *** ---!>\n") 

315 if main is not None: 

316 main.report_issue(body=issue_body, title=title, 

317 open_webpage=True) 

318 else: 

319 pass

339 def _contents_changed(self): 

340 """Activate submit_btn.""" 

341 if not self.require_minimum_length: 

342 return 

343 desc_chars = (len(self.input_description.toPlainText()) - 

344 self.initial_chars) 

345 if desc_chars < self.description_minimum_length: 

346 self.desc_chars_label.setText( 

347 u"{} {}".format(self.description_minimum_length - desc_chars, 

348 _("more characters to go..."))) 

349 else: 

350 self.desc_chars_label.setText(_("Description complete; thanks!")) 

351 

352 title_chars = len(self.title.text()) 

353 if title_chars < TITLE_MIN_CHARS: 

354 self.title_chars_label.setText( 

355 u"{} {}".format(TITLE_MIN_CHARS - title_chars, 

356 _("more characters to go..."))) 

357 else: 

358 self.title_chars_label.setText(_("Title complete; thanks!")) 

359 

360 submission_enabled = (desc_chars >= self.description_minimum_length and 

361 title_chars >= TITLE_MIN_CHARS) 

362 self.submit_btn.setEnabled(submission_enabled)

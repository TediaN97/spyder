29 self.clear()

44 # Spyder plugins 

45 spyder_plugins = QTreeWidgetItem([_("Spyder plugins")]) 

46 spyder_plugins.setFont(0, font) 

47 

48 self.addTopLevelItems([mandatory_item, optional_item, spyder_plugins]) 

49 

50 for dependency in sorted(dependencies, 

51 key=lambda x: x.modname.lower()): 

52 item = QTreeWidgetItem([dependency.modname, 

53 dependency.package_name, 

54 dependency.required_version, 

55 dependency.installed_version, 

56 dependency.features]) 

57 # Format content 

58 if dependency.check(): 

59 item.setIcon(0, ima.icon('dependency_ok')) 

60 elif dependency.kind == OPTIONAL: 

61 item.setIcon(0, ima.icon('dependency_warning')) 

62 item.setForeground(2, QColor('#ff6a00')) 

63 else: 

64 item.setIcon(0, ima.icon('dependency_error')) 

65 item.setForeground(2, QColor(Qt.darkRed)) 

66 

67 # Add to tree 

68 if dependency.kind == OPTIONAL: 

69 optional_item.addChild(item) 

70 elif dependency.kind == PLUGIN: 

71 spyder_plugins.addChild(item) 

72 else: 

73 mandatory_item.addChild(item)

15def unmatched_quotes_in_line(text): 

16 """Return whether a string has open quotes. 

17 

18 This simply counts whether the number of quote characters of either 

19 type in the string is odd. 

20 

21 Take from the IPython project (in IPython/core/completer.py in v0.13) 

22 Spyder team: Add some changes to deal with escaped quotes 

23 

24 - Copyright (C) 2008-2011 IPython Development Team 

25 - Copyright (C) 2001-2007 Fernando Perez. <fperez@colorado.edu> 

26 - Copyright (C) 2001 Python Software Foundation, www.python.org 

27 

28 Distributed under the terms of the BSD License. 

29 """ 

30 # We check " first, then ', so complex cases with nested quotes will 

31 # get the " to take precedence. 

32 text = text.replace("\\'", "") 

33 text = text.replace('\\"', '') 

34 if text.count('"') % 2: 

35 return '"' 

36 elif text.count("'") % 2: 

37 return "'" 

38 else: 

39 return ''

42class CloseQuotesExtension(EditorExtension): 

43 """Editor Extension for insert closing quotes automatically.""" 

44 

45 def on_state_changed(self, state): 

46 """Connect/disconnect sig_key_pressed signal.""" 

47 if state: 

48 self.editor.sig_key_pressed.connect(self._on_key_pressed) 

49 else: 

50 self.editor.sig_key_pressed.disconnect(self._on_key_pressed) 

51 

52 def _on_key_pressed(self, event): 

53 if event.isAccepted(): 

54 return 

55 

56 # It is necessary to use here the text of the event and not the key 

57 # to avoid issues with international keyboards. 

58 # See spyder-ide/spyder#9814 

59 char = event.text() 

60 if char in ('"', '\'') and self.enabled: 

61 self.editor.completion_widget.hide() 

62 self._autoinsert_quotes(char) 

63 self.editor.document_did_change() 

64 event.accept()

66 def _autoinsert_quotes(self, char): 

67 """Control how to automatically insert quotes in various situations.""" 

68 line_text = self.editor.get_text('sol', 'eol') 

69 line_to_cursor = self.editor.get_text('sol', 'cursor') 

70 cursor = self.editor.textCursor() 

71 last_three = self.editor.get_text('sol', 'cursor')[-3:] 

72 last_two = self.editor.get_text('sol', 'cursor')[-2:] 

73 trailing_text = self.editor.get_text('cursor', 'eol').strip() 

74 

75 if self.editor.has_selected_text(): 

76 text = self.editor.get_selected_text() 

77 self.editor.insert_text("{0}{1}{0}".format(char, text)) 

78 # keep text selected, for inserting multiple quotes 

79 cursor.movePosition(QTextCursor.Left, QTextCursor.MoveAnchor, 1) 

80 cursor.movePosition(QTextCursor.Left, QTextCursor.KeepAnchor, 

81 len(text)) 

82 self.editor.setTextCursor(cursor) 

83 elif self.editor.in_comment(): 

84 self.editor.insert_text(char) 

85 elif (len(trailing_text) > 0 and 

86 not unmatched_quotes_in_line(line_to_cursor) == char and 

87 not trailing_text[0] in (',', ':', ';', ')', ']', '}')): 

88 self.editor.insert_text(char) 

89 elif (unmatched_quotes_in_line(line_text) and 

90 (not last_three == 3*char)): 

91 self.editor.insert_text(char) 

92 # Move to the right if we are before a quote 

93 elif self.editor.next_char() == char: 

94 cursor.movePosition(QTextCursor.NextCharacter, 

95 QTextCursor.KeepAnchor, 1) 

96 cursor.clearSelection() 

97 self.editor.setTextCursor(cursor) 

98 # Automatic insertion of triple double quotes (for docstrings) 

99 elif last_three == 3*char: 

100 self.editor.insert_text(3*char) 

101 cursor = self.editor.textCursor() 

102 cursor.movePosition(QTextCursor.PreviousCharacter, 

103 QTextCursor.KeepAnchor, 3) 

104 cursor.clearSelection() 

105 self.editor.setTextCursor(cursor) 

106 # If last two chars are quotes, just insert one more because most 

107 # probably the user wants to write a docstring 

108 elif last_two == 2*char: 

109 self.editor.insert_text(char) 

110 self.editor.delayed_popup_docstring() 

111 # Automatic insertion of quotes 

112 else: 

113 self.editor.insert_text(2*char) 

114 cursor = self.editor.textCursor() 

115 cursor.movePosition(QTextCursor.PreviousCharacter) 

116 self.editor.setTextCursor(cursor)

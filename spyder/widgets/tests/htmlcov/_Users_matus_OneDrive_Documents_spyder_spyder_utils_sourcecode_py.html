25def get_eol_chars(text): 

26 """ 

27 Get text EOL characters. 

28 

29 If None is found, return the eol based on the operatin system. 

30 """ 

31 for eol_chars, _os_name in EOL_CHARS: 

32 if text.find(eol_chars) > -1: 

33 break 

34 else: 

35 if os.name == 'nt': 

36 eol_chars = "\r\n" 

37 elif sys.platform.startswith('linux'): 

38 eol_chars = "\n" 

39 elif sys.platform == 'darwin': 

40 eol_chars = "\r" 

41 else: 

42 eol_chars = "\n" 

43 

44 return eol_chars

96def get_primary_at(source_code, offset, retry=True): 

97 """Return Python object in *source_code* at *offset* 

98 Periods to the left of the cursor are carried forward 

99 e.g. 'functools.par^tial' would yield 'functools.partial' 

100 Retry prevents infinite recursion: retry only once 

101 """ 

102 obj = '' 

103 left = re.split(r"[^0-9a-zA-Z_.]", source_code[:offset]) 

104 if left and left[-1]: 

105 obj = left[-1] 

106 right = re.split(r"\W", source_code[offset:]) 

107 if right and right[0]: 

108 obj += right[0] 

109 if obj and obj[0].isdigit(): 

110 obj = '' 

111 # account for opening chars with no text to the right 

112 if not obj and retry and offset and source_code[offset - 1] in '([.': 

113 return get_primary_at(source_code, offset - 1, retry=False) 

114 return obj

127def get_identifiers(source_code): 

128 '''Split source code into python identifier-like tokens''' 

129 tokens = set(re.split(r"[^0-9a-zA-Z_.]", source_code)) 

130 valid = re.compile(r'[a-zA-Z_]') 

131 return [token for token in tokens if re.match(valid, token)]

133def path_components(path): 

134 """ 

135 Return the individual components of a given file path 

136 string (for the local operating system). 

137 

138 Taken from https://stackoverflow.com/q/21498939/438386 

139 """ 

140 components = [] 

141 # The loop guarantees that the returned components can be 

142 # os.path.joined with the path separator and point to the same 

143 # location: 

144 while True: 

145 (new_path, tail) = os.path.split(path) # Works on any platform 

146 components.append(tail) 

147 if new_path == path: # Root (including drive, on Windows) reached 

148 break 

149 path = new_path 

150 components.append(new_path) 

151 components.reverse() # First component first 

152 return components

154def differentiate_prefix(path_components0, path_components1): 

155 """ 

156 Return the differentiated prefix of the given two iterables. 

157 

158 Taken from https://stackoverflow.com/q/21498939/438386 

159 """ 

160 longest_prefix = [] 

161 root_comparison = False 

162 common_elmt = None 

163 for index, (elmt0, elmt1) in enumerate(zip(path_components0, path_components1)): 

164 if elmt0 != elmt1: 

165 if index == 2: 

166 root_comparison = True 

167 break 

168 else: 

169 common_elmt = elmt0 

170 longest_prefix.append(elmt0) 

171 file_name_length = len(path_components0[len(path_components0) - 1]) 

172 path_0 = os.path.join(*path_components0)[:-file_name_length - 1] 

173 if len(longest_prefix) > 0: 

174 longest_path_prefix = os.path.join(*longest_prefix) 

175 longest_prefix_length = len(longest_path_prefix) + 1 

176 if path_0[longest_prefix_length:] != '' and not root_comparison: 

177 path_0_components = path_components(path_0[longest_prefix_length:]) 

178 if path_0_components[0] == ''and path_0_components[1] == ''and len( 

179 path_0[longest_prefix_length:]) > 20: 

180 path_0_components.insert(2, common_elmt) 

181 path_0 = os.path.join(*path_0_components) 

182 else: 

183 path_0 = path_0[longest_prefix_length:] 

184 elif not root_comparison: 

185 path_0 = common_elmt 

186 elif sys.platform.startswith('linux') and path_0 == '': 

187 path_0 = '/' 

188 return path_0

190def disambiguate_fname(files_path_list, filename): 

191 """Get tab title without ambiguation.""" 

192 fname = os.path.basename(filename) 

193 same_name_files = get_same_name_files(files_path_list, fname) 

194 if len(same_name_files) > 1: 

195 compare_path = shortest_path(same_name_files) 

196 if compare_path == filename: 

197 same_name_files.remove(path_components(filename)) 

198 compare_path = shortest_path(same_name_files) 

199 diff_path = differentiate_prefix(path_components(filename), 

200 path_components(compare_path)) 

201 diff_path_length = len(diff_path) 

202 path_component = path_components(diff_path) 

203 if (diff_path_length > 20 and len(path_component) > 2): 

204 if path_component[0] != '/' and path_component[0] != '': 

205 path_component = [path_component[0], '...', 

206 path_component[-1]] 

207 else: 

208 path_component = [path_component[2], '...', 

209 path_component[-1]] 

210 diff_path = os.path.join(*path_component) 

211 fname = fname + " - " + diff_path 

212 return fname

222def shortest_path(files_path_list): 

223 """Shortest path between files in the list."""

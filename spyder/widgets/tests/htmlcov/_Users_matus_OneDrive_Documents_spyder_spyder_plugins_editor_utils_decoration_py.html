27# Timeout to avoid almost simultaneous calls to update decorations, which 

28# introduces a lot of sluggishness in the editor. 

29UPDATE_TIMEOUT = 15 # miliseconds 

30 

31 

32class TextDecorationsManager(Manager, QObject): 

33 """ 

34 Manages the collection of TextDecoration that have been set on the editor 

35 widget. 

36 """ 

37 def __init__(self, editor): 

38 super(TextDecorationsManager, self).__init__(editor) 

39 QObject.__init__(self, None) 

40 self._decorations = [] 

41 

42 # Timer to not constantly update decorations. 

43 self.update_timer = QTimer(self) 

44 self.update_timer.setSingleShot(True) 

45 self.update_timer.setInterval(UPDATE_TIMEOUT) 

46 self.update_timer.timeout.connect( 

47 self._update) 

48 

49 def add(self, decorations): 

50 """ 

51 Add text decorations on a CodeEditor instance. 

52 

53 Don't add duplicated decorations, and order decorations according 

54 draw_order and the size of the selection. 

55 

56 Args: 

57 decorations (sourcecode.api.TextDecoration) (could be a list) 

58 Returns: 

59 int: Amount of decorations added. 

60 """ 

61 added = 0 

62 if isinstance(decorations, list): 

63 not_repeated = set(decorations) - set(self._decorations) 

64 self._decorations.extend(list(not_repeated)) 

65 added = len(not_repeated) 

66 elif decorations not in self._decorations: 

67 self._decorations.append(decorations) 

68 added = 1 

69 

70 if added > 0: 

71 self._order_decorations() 

72 self.update() 

73 return added 

74 

75 def remove(self, decoration): 

76 """ 

77 Removes a text decoration from the editor. 

78 

79 :param decoration: Text decoration to remove 

80 :type decoration: spyder.api.TextDecoration 

81 update: Bool: should the decorations be updated immediately? 

82 Set to False to avoid updating several times while removing 

83 several decorations 

84 """ 

85 try: 

86 self._decorations.remove(decoration) 

87 self.update() 

88 return True 

89 except ValueError: 

90 return False 

91 

92 def clear(self): 

93 """Removes all text decoration from the editor.""" 

94 self._decorations[:] = [] 

95 self.update() 

96 

97 def update(self): 

98 """ 

99 Update decorations. 

100 

101 This starts a timer to update decorations only after 

102 UPDATE_TIMEOUT has passed. That avoids multiple calls to 

103 _update in a very short amount of time. 

104 """ 

105 self.update_timer.start() 

106 

107 @Slot() 

108 def _update(self): 

109 """Update editor extra selections with added decorations. 

110 

111 NOTE: Update TextDecorations to use editor font, using a different 

112 font family and point size could cause unwanted behaviors. 

113 """ 

114 try: 

115 font = self.editor.font() 

116 

117 # Get the current visible block numbers 

118 first, last = self.editor.get_buffer_block_numbers() 

119 

120 # Update visible decorations 

121 visible_decorations = [] 

122 for decoration in self._decorations: 

123 block_nb = decoration.cursor.block().blockNumber() 

124 if (first <= block_nb <= last or 

125 decoration.kind == 'current_cell'): 

126 visible_decorations.append(decoration) 

127 try: 

128 decoration.format.setFont( 

129 font, QTextCharFormat.FontPropertiesSpecifiedOnly) 

130 except (TypeError, AttributeError): # Qt < 5.3 

131 decoration.format.setFontFamily(font.family()) 

132 decoration.format.setFontPointSize(font.pointSize()) 

133 

134 self.editor.setExtraSelections(visible_decorations) 

135 except RuntimeError: 

136 # This is needed to fix spyder-ide/spyder#9173. 

137 return 

138 

139 def __iter__(self): 

140 return iter(self._decorations) 

141 

142 def __len__(self): 

143 return len(self._decorations) 

144 

145 def _order_decorations(self): 

146 """Order decorations according draw_order and size of selection. 

147 

148 Highest draw_order will appear on top of the lowest values. 

149 

150 If draw_order is equal,smaller selections are draw in top of 

151 bigger selections. 

152 """ 

153 def order_function(sel): 

154 end = sel.cursor.selectionEnd() 

155 start = sel.cursor.selectionStart() 

156 return sel.draw_order, -(end - start) 

157 

158 self._decorations = sorted(self._decorations, 

159 key=order_function)

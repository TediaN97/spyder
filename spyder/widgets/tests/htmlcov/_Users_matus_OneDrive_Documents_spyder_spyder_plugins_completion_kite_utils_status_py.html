26def check_if_kite_installed(): 

27 """Detect if kite is installed and return the installation path.""" 

28 path = '' 

29 if os.name == 'nt': 

30 path = 'C:\\Program Files\\Kite\\kited.exe' 

31 elif sys.platform.startswith('linux'): 

32 path = osp.expanduser('~/.local/share/kite/kited') 

33 elif sys.platform == 'darwin': 

34 path = locate_kite_darwin() 

35 return osp.exists(osp.realpath(path)), path

38def check_if_kite_running(): 

39 """Detect if kite is running.""" 

40 running = False 

41 try: 

42 for proc in psutil.process_iter(attrs=['pid', 'name', 'username', 

43 'status']): 

44 if is_proc_kite(proc): 

45 logger.debug('Kite process already ' 

46 'running with PID {0}'.format(proc.pid)) 

47 running = True 

48 break 

49 except OSError: 

50 # Needed to handle a possible WinError 0. See spyder-ide/spyder#12510 

51 pass 

52 return running

55def locate_kite_darwin(): 

56 """ 

57 Looks up where Kite.app is installed on macOS systems. The bundle ID 

58 is checked first and if nothing is found or an error occurs, the 

59 default path is used. 

60 """ 

61 default_path = '/Applications/Kite.app' 

62 path = '' 

63 try: 

64 out = subprocess.check_output( 

65 ['mdfind', 'kMDItemCFBundleIdentifier="com.kite.Kite"']) 

66 installed = len(out) > 0 

67 path = (out.decode('utf-8', 'replace').strip().split('\n')[0] 

68 if installed else default_path) 

69 except (subprocess.CalledProcessError, UnicodeDecodeError): 

70 # Use the default path 

71 path = default_path 

72 finally: 

73 return path

76def is_proc_kite(proc): 

77 try: 

78 # This is raising `ZombieProcess: psutil.ZombieProcess` on OSX 

79 # if kite is not running. 

80 name = proc.name() 

81 except Exception: 

82 name = '' 

83 

84 if os.name == 'nt' or sys.platform.startswith('linux'): 

85 is_kite = 'kited' in name and proc.status() != 'zombie' 

86 else: 

87 is_kite = 'Kite' == name 

88 

89 return is_kite

92def status(extra_status=''): 

93 """Kite completions status: not installed, ready, not running.""" 

94 kite_installed, _ = check_if_kite_installed() 

95 if not kite_installed: 

96 return NOT_INSTALLED + extra_status 

97 elif check_if_kite_running(): 

98 return RUNNING + extra_status 

99 else: 

100 return NOT_RUNNING + extra_status

33def drift_color(base_color, factor=110): 

34 """ 

35 Return color that is lighter or darker than the base color. 

36 

37 If base_color.lightness is higher than 128, the returned color is darker 

38 otherwise is is lighter. 

39 

40 :param base_color: The base color to drift from 

41 ;:param factor: drift factor (%) 

42 :return A lighter or darker color. 

43 """ 

44 base_color = QColor(base_color) 

45 if base_color.lightness() > 128: 

46 return base_color.darker(factor) 

47 else: 

48 if base_color == QColor('#000000'): 

49 return drift_color(QColor('#101010'), factor + 20) 

50 else: 

51 return base_color.lighter(factor + 10)

109class DelayJobRunner(object): 

110 """ 

111 Utility class for running job after a certain delay. 

112 If a new request is made during this delay, the previous request is dropped 

113 and the timer is restarted for the new request. 

114 

115 We use this to implement a cooldown effect that prevents jobs from being 

116 executed while the IDE is not idle. 

117 

118 A job is a simple callable. 

119 """ 

120 def __init__(self, delay=500): 

121 """ 

122 :param delay: Delay to wait before running the job. This delay applies 

123 to all requests and cannot be changed afterwards. 

124 """ 

125 self._timer = QTimer() 

126 self.delay = delay 

127 self._timer.timeout.connect(self._exec_requested_job) 

128 self._args = [] 

129 self._kwargs = {} 

130 self._job = lambda x: None 

131 

132 def request_job(self, job, *args, **kwargs): 

133 """ 

134 Request a job execution. 

135 

136 The job will be executed after the delay specified in the 

137 DelayJobRunner contructor elapsed if no other job is requested until 

138 then. 

139 

140 :param job: job. 

141 :type job: callable 

142 :param args: job's position arguments 

143 :param kwargs: job's keyworded arguments 

144 """ 

145 self.cancel_requests() 

146 self._job = job 

147 self._args = args 

148 self._kwargs = kwargs 

149 self._timer.start(self.delay) 

150 

151 def cancel_requests(self): 

152 """Cancels pending requests.""" 

153 self._timer.stop() 

154 self._job = None 

155 self._args = None 

156 self._kwargs = None 

157 

158 def _exec_requested_job(self): 

159 """Execute the requested job after the timer has timeout.""" 

160 self._timer.stop() 

161 self._job(*self._args, **self._kwargs)

239 def word_under_cursor(self, select_whole_word=False, text_cursor=None): 

240 """ 

241 Gets the word under cursor using the separators defined by 

242 :attr:`spyder.plugins.editor.widgets.codeeditor.CodeEditor.word_separators`. 

243 

244 FIXME: This is not working because CodeEditor have no attribute 

245 word_separators 

246 

247 .. note: Instead of returning the word string, this function returns 

248 a QTextCursor, that way you may get more information than just the 

249 string. To get the word, just call ``selectedText`` on the returned 

250 value. 

251 

252 :param select_whole_word: If set to true the whole word is selected, 

253 else the selection stops at the cursor position. 

254 :param text_cursor: Optional custom text cursor (e.g. from a 

255 QTextDocument clone) 

256 :returns: The QTextCursor that contains the selected word. 

257 """ 

258 editor = self._editor 

259 if not text_cursor: 

260 text_cursor = editor.textCursor() 

261 word_separators = editor.word_separators 

262 end_pos = start_pos = text_cursor.position() 

263 # select char by char until we are at the original cursor position. 

264 while not text_cursor.atStart(): 

265 text_cursor.movePosition( 

266 text_cursor.Left, text_cursor.KeepAnchor, 1) 

267 try: 

268 char = text_cursor.selectedText()[0] 

269 word_separators = editor.word_separators 

270 selected_txt = text_cursor.selectedText() 

271 if (selected_txt in word_separators and 

272 (selected_txt != "n" and selected_txt != "t") or 

273 char.isspace()): 

274 break # start boundary found 

275 except IndexError: 

276 break # nothing selectable 

277 start_pos = text_cursor.position() 

278 text_cursor.setPosition(start_pos) 

279 if select_whole_word: 

280 # select the resot of the word 

281 text_cursor.setPosition(end_pos) 

282 while not text_cursor.atEnd(): 

283 text_cursor.movePosition(text_cursor.Right, 

284 text_cursor.KeepAnchor, 1) 

285 char = text_cursor.selectedText()[0] 

286 selected_txt = text_cursor.selectedText() 

287 if (selected_txt in word_separators and 

288 (selected_txt != "n" and selected_txt != "t") or 

289 char.isspace()): 

290 break # end boundary found 

291 end_pos = text_cursor.position() 

292 text_cursor.setPosition(end_pos) 

293 # now that we habe the boundaries, we can select the text 

294 text_cursor.setPosition(start_pos) 

295 text_cursor.setPosition(end_pos, text_cursor.KeepAnchor) 

296 return text_cursor

397 def clean_document(self): 

398 """ 

399 Removes trailing whitespaces and ensure one single blank line at the 

400 end of the QTextDocument. 

401 

402 FIXME: It was deprecated in pyqode, maybe It should be deleted 

403 """ 

404 editor = self._editor 

405 value = editor.verticalScrollBar().value() 

406 pos = self.cursor_position() 

407 editor.textCursor().beginEditBlock() 

408 

409 # cleanup whitespaces 

410 editor._cleaning = True 

411 eaten = 0 

412 removed = set() 

413 for line in editor._modified_lines: 

414 # parse line before and line after modified line (for cases where 

415 # key_delete or key_return has been pressed) 

416 for j in range(-1, 2): 

417 # skip current line 

418 if line + j != pos[0]: 

419 if line + j >= 0: 

420 txt = self.line_text(line + j) 

421 stxt = txt.rstrip() 

422 if txt != stxt: 

423 self.set_line_text(line + j, stxt) 

424 removed.add(line + j) 

425 editor._modified_lines -= removed 

426 

427 # ensure there is only one blank line left at the end of the file 

428 i = self.line_count() 

429 while i: 

430 line = self.line_text(i - 1) 

431 if line.strip(): 

432 break 

433 self.remove_last_line() 

434 i -= 1 

435 if self.line_text(self.line_count() - 1): 

436 editor.appendPlainText('') 

437 

438 # restore cursor and scrollbars 

439 text_cursor = editor.textCursor() 

440 doc = editor.document() 

441 assert isinstance(doc, QTextDocument) 

442 text_cursor = self._move_cursor_to(pos[0]) 

443 text_cursor.movePosition(text_cursor.StartOfLine, 

444 text_cursor.MoveAnchor) 

445 cpos = text_cursor.position() 

446 text_cursor.select(text_cursor.LineUnderCursor) 

447 if text_cursor.selectedText(): 

448 text_cursor.setPosition(cpos) 

449 offset = pos[1] - eaten 

450 text_cursor.movePosition(text_cursor.Right, text_cursor.MoveAnchor, 

451 offset) 

452 else: 

453 text_cursor.setPosition(cpos) 

454 editor.setTextCursor(text_cursor) 

455 editor.verticalScrollBar().setValue(value) 

456 

457 text_cursor.endEditBlock() 

458 editor._cleaning = False 

459 editor.document_did_change()

481 def select_lines(self, start=0, end=-1, apply_selection=True): 

482 """ 

483 Selects entire lines between start and end line numbers. 

484 

485 This functions apply the selection and returns the text cursor that 

486 contains the selection. 

487 

488 Optionally it is possible to prevent the selection from being applied 

489 on the code editor widget by setting ``apply_selection`` to False. 

490 

491 :param start: Start line number (0 based) 

492 :param end: End line number (0 based). Use -1 to select up to the 

493 end of the document 

494 :param apply_selection: True to apply the selection before returning 

495 the QTextCursor. 

496 :returns: A QTextCursor that holds the requested selection 

497 """ 

498 editor = self._editor 

499 if end == -1: 

500 end = self.line_count() - 1 

501 if start < 0: 

502 start = 0 

503 text_cursor = self._move_cursor_to(start) 

504 if end > start: # Going down 

505 text_cursor.movePosition(text_cursor.Down, 

506 text_cursor.KeepAnchor, end - start) 

507 text_cursor.movePosition(text_cursor.EndOfLine, 

508 text_cursor.KeepAnchor) 

509 elif end < start: # going up 

510 # don't miss end of line ! 

511 text_cursor.movePosition(text_cursor.EndOfLine, 

512 text_cursor.MoveAnchor) 

513 text_cursor.movePosition(text_cursor.Up, 

514 text_cursor.KeepAnchor, start - end) 

515 text_cursor.movePosition(text_cursor.StartOfLine, 

516 text_cursor.KeepAnchor) 

517 else: 

518 text_cursor.movePosition(text_cursor.EndOfLine, 

519 text_cursor.KeepAnchor) 

520 if apply_selection: 

521 editor.setTextCursor(text_cursor) 

522 return text_cursor

754 """

838 closed[opposite] -= 1

840 else:

842 start_pos = cursor.position() + 1 

843 stop = True

898class TextBlockHelper(object): 

899 """ 

900 Helps retrieving the various part of the user state bitmask. 

901 

902 This helper should be used to replace calls to 

903 ``QTextBlock.setUserState``/``QTextBlock.getUserState`` as well as 

904 ``QSyntaxHighlighter.setCurrentBlockState``/ 

905 ``QSyntaxHighlighter.currentBlockState`` and 

906 ``QSyntaxHighlighter.previousBlockState``. 

907 

908 The bitmask is made up of the following fields: 

909 

910 - bit0 -> bit26: User state (for syntax highlighting) 

911 - bit26: fold trigger state 

912 - bit27-bit29: fold level (8 level max) 

913 - bit30: fold trigger flag 

914 

915 - bit0 -> bit15: 16 bits for syntax highlighter user state ( 

916 for syntax highlighting) 

917 - bit16-bit25: 10 bits for the fold level (1024 levels) 

918 - bit26: 1 bit for the fold trigger flag (trigger or not trigger) 

919 - bit27: 1 bit for the fold trigger state (expanded/collapsed) 

920 

921 """ 

922 @staticmethod 

923 def get_state(block): 

924 """ 

925 Gets the user state, generally used for syntax highlighting. 

926 :param block: block to access 

927 :return: The block state 

928 

929 """ 

930 if block is None: 

931 return -1 

932 state = block.userState() 

933 if state == -1: 

934 return state 

935 return state & 0x0000FFFF 

936 

937 @staticmethod 

938 def set_state(block, state): 

939 """ 

940 Sets the user state, generally used for syntax highlighting. 

941 

942 :param block: block to modify 

943 :param state: new state value. 

944 :return: 

945 """ 

946 if block is None: 

947 return 

948 user_state = block.userState() 

949 if user_state == -1: 

950 user_state = 0 

951 higher_part = user_state & 0x7FFF0000 

952 state &= 0x0000FFFF 

953 state |= higher_part 

954 block.setUserState(state) 

955 

956 @staticmethod 

957 def get_fold_lvl(block): 

958 """ 

959 Gets the block fold level. 

960 

961 :param block: block to access. 

962 :returns: The block fold level 

963 """ 

964 if block is None: 

965 return 0 

966 state = block.userState() 

967 if state == -1: 

968 state = 0 

969 return (state & 0x03FF0000) >> 16 

970 

971 @staticmethod 

972 def set_fold_lvl(block, val): 

973 """ 

974 Sets the block fold level. 

975 

976 :param block: block to modify 

977 :param val: The new fold level [0-7] 

978 """ 

979 if block is None: 

980 return 

981 state = block.userState() 

982 if state == -1: 

983 state = 0 

984 if val >= 0x3FF: 

985 val = 0x3FF 

986 state &= 0x7C00FFFF 

987 state |= val << 16 

988 block.setUserState(state) 

989 

990 @staticmethod 

991 def is_fold_trigger(block): 

992 """ 

993 Checks if the block is a fold trigger. 

994 

995 :param block: block to check 

996 :return: True if the block is a fold trigger (represented as a node in 

997 the fold panel) 

998 """ 

999 if block is None: 

1000 return False 

1001 state = block.userState() 

1002 if state == -1: 

1003 state = 0 

1004 return bool(state & 0x04000000)

1006 @staticmethod 

1007 def set_fold_trigger(block, val):

1010 trigger).

1012 :param block: block to set

1014 """

72def _binary_stdio(): 

73 """Construct binary stdio streams (not text mode). 

74 

75 This seems to be different for Window/Unix Python2/3, so going by: 

76 https://stackoverflow.com/questions/2850893/reading-binary-data-from-stdin 

77 """ 

78 PY3K = sys.version_info >= (3, 0) 

79 

80 if PY3K: 

81 # pylint: disable=no-member 

82 stdin, stdout = sys.stdin.buffer, sys.stdout.buffer 

83 else: 

84 # Python 2 on Windows opens sys.stdin in text mode, and 

85 # binary data that read from it becomes corrupted on \r\n 

86 if sys.platform == "win32": 

87 # set sys.stdin to binary mode 

88 # pylint: disable=no-member,import-error 

89 import os 

90 import msvcrt 

91 msvcrt.setmode(sys.stdin.fileno(), os.O_BINARY) 

92 msvcrt.setmode(sys.stdout.fileno(), os.O_BINARY) 

93 stdin, stdout = sys.stdin, sys.stdout 

94 

95 return stdin, stdout
